def cross_tab(dataframe, col, columns_list):
    new = []
    for c in columns_list:
        new.append(pd.crosstab(dataframe[col], dataframe[c], margins = True, margins_name = 'Total'))
    new_df = pd.concat(new, axis = 1, keys = columns_list)
    return new_df
